#### AppUpdate
```
    /// 检测更新 请求返回数据是固定的
    AppUpdate.checkUpdate(
      context,
      url: '/home/app',
      appId: 'io.cordova.maixiaobu',
    );

    /// 要求返回的数据格式
    {
      'data': {
        /// 是否更新
        bool hasUpdate;

        /// 是否强制更新
        bool isIgnorable;

        /// 是否跳转appStore
        bool isAppStore;

        /// app版本号
        String versionCode;

        /// app大小
        String apkSize;

        /// 下载地址
        String downloadUrl;

        /// 更新内容
        String updateContent;

        /// 更新类型
        /// normal 全量升级
        ///
        /// hotUpgrade 热更新
        ///
        /// incrementUpgrade 增量升级
        String type;
      },
      'code': 200,
      'msg': '请求成功'
    }
```
